# 📘 1. Introduction

This document provides a professional **Standard Operating Procedure (SOP)** for extracting:

* 🖥️ Hostname
* 🔗 MAC Address
* 🌐 IP Address

from a Sophos Firewall configuration backup.

In enterprise environments, DHCP static reservations are commonly stored inside the firewall configuration. Extracting this information helps IT teams maintain accurate:

* 📋 Network documentation
* 🗂️ Asset inventories
* 🔍 Device audit records
* 🔄 Migration references

This procedure explains how to:

1. Export the Sophos Firewall configuration
2. Extract the `Entities.xml` file
3. Parse the XML data using Python
4. Generate a structured CSV report

> ✅ Compatible with all Sophos Firewall OS (SFOS) versions.

## 🔹 Step 2 — Paste the Python Script

Copy and paste the following script into the file.

```python id="9nqnn7"
# entities_to_csv.py

import xml.etree.ElementTree as ET
import csv
from pathlib import Path

xml_path = "Entities.xml"
csv_path = "reservations.csv"

def strip_ns(tag: str) -> str:
    """Return tag name without XML namespace, lowercased."""
    return tag.split('}')[-1].lower() if '}' in tag else tag.lower()

def get_first(node, names):
    """
    Return text for the first matching attribute or child tag name.
    Handles namespaces and case-insensitive matching.
    """

    # Check attributes
    for k, v in node.attrib.items():
        if k.lower() in names and v and v.strip():
            return v.strip()

    # Check child elements
    for child in list(node):
        if strip_ns(child.tag) in names:
            txt = (child.text or '').strip()
            if txt:
                return txt

    return None

# Common key variants seen in SFOS XML exports
HOST_KEYS = {"hostname", "host", "host_name", "host-name", "name"}

MAC_KEYS = {
    "mac",
    "macaddress",
    "mac_addr",
    "mac-address",
    "macaddress1"
}

IP_KEYS = {
    "ip",
    "ipaddress",
    "ip-address",
    "ipv4",
    "ipv4address"
}

# Node names that often represent reservation containers
LIKELY_NODE_NAMES = {
    "ipmacbinding",
    "ip-mac-binding",
    "ipmacmap",
    "staticbinding",
    "static-binding",
    "staticipmac",
    "static-ip-mac",
    "dhcpstaticmapping",
    "reservation",
    "mapping"
}

tree = ET.parse(xml_path)
root = tree.getroot()

rows = []
hits = 0

for node in root.iter():

    tag = strip_ns(node.tag)

    looks_like_binding = any(key in tag for key in LIKELY_NODE_NAMES)

    host = get_first(node, HOST_KEYS)
    mac = get_first(node, MAC_KEYS)
    ip = get_first(node, IP_KEYS)

    if looks_like_binding or (mac and ip):

        if mac and ip:
            rows.append([host or "", mac.lower(), ip])
            hits += 1

# Remove duplicate entries
seen = set()
unique = []

for h, m, i in rows:

    k = (h, m, i)

    if k not in seen:
        seen.add(k)
        unique.append([h, m, i])

# Export CSV
with open(csv_path, "w", newline="", encoding="utf-8") as f:

    w = csv.writer(f)

    w.writerow(["hostname", "mac", "ip"])
    w.writerows(unique)

print(f"Found {hits} potential bindings.")
print(f"Wrote {len(unique)} unique rows to {csv_path}")
print(f"Output file: {Path(csv_path).resolve()}")
```
